}

bool push(tpilha * pp, int dado){
	tpilha novo = (tno *)malloc(sizeof(tno))
	novo->dado = dado;
	novo->prox = *pp;
	*pp = novo;
	
	return true;
}

}

int main(int argc, char *argv[]) {
	
	tpilha p;
	int escolha;
	
	criarPilha(&p);
	
	do{
		printf("Insira o que deseja fazer: \n1)Criar pilha\n2)Inserir na pilha\n3)Retirar da pilha\n4)Sair");
